#include <stdio.h>

#define swap_char(x,y) {*x ^= *y; *y ^= *x; *x ^= *y;}

int pivot(int *str, int S, int E)
{
   int pIndex = S;
   int i;
   int pivot = str[E];

   for (i = S; i < E; i++)
   { 
      if (str[i] <= pivot)
      {
         swap_char((str + i), (str + pIndex))
         ++pIndex;
      }
   }

   swap_char((str + pIndex), (str + E))

   return(pIndex);
}

void quickSort(int *str, int S, int E)
{
   if (S < E)
   {
      int pivot = pivot(str, S, E);

      quickSort(str, S, pivot - 1);
      quickSort(str, pivot + 1, E);
   }
}
